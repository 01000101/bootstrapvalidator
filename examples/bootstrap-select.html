Play with Bootstrap-select

Use with the Bootstrap select plugin.

The Favorite color option requires to choose 2-4 colors you like most.

The Language option requires to choose your native language.

{% highlight javascript hl_lines="7 14 18" %} $(document).ready(function() { $('#bSelectForm') .find('[name="colors"]') .selectpicker() .change(function(e) { // revalidate the color when it is changed $('#bSelectForm').bootstrapValidator('revalidateField', 'colors'); }) .end() .find('[name="language"]') .selectpicker() .change(function(e) { // revalidate the language when it is changed $('#bSelectForm').bootstrapValidator('revalidateField', 'language'); }) .end() .bootstrapValidator({ excluded: ':disabled', feedbackIcons: { valid: 'glyphicon glyphicon-ok', invalid: 'glyphicon glyphicon-remove', validating: 'glyphicon glyphicon-refresh' }, fields: { colors: { validators: { callback: { message: 'Please choose 2-4 colors you like most', callback: function(value, validator) { // Get the selected options var options = validator.getFieldElements('colors').val(); return (options != null && options.length >= 2 && options.length <= 4); } } } }, language: { validators: { notEmpty: { message: 'Please select one language you can speak.' } } } } }) .on('success.form.bv', function(e) { // Prevent form submission e.preventDefault(); }); }); {% endhighlight %}
